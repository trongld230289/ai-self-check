// extension.ts for AIDA Core
export function activate(context: vscode.ExtensionContext) {
    // Register workflow commands
    const runWorkflow = vscode.commands.registerCommand(
        'aida.runWorkflow', 
        async (workflowName: string) => {
            const workflow = await loadWorkflow(workflowName);
            await executeWorkflowSteps(workflow);
        }
    );
    
    // Listen for sub-extension events
    const eventListener = vscode.workspace.onDidChangeTextDocument((e) => {
        if (e.document.uri.path.includes('/DOCS/')) {
            handleDocumentChange(e);
        }
    });
    
    context.subscriptions.push(runWorkflow, eventListener);
}

async function executeWorkflowSteps(workflow: Workflow) {
    for (const step of workflow.steps) {
        await vscode.commands.executeCommand(
            `aida-${step.role}.${step.action}`, 
            step.params
        );
        await waitForCompletion(step);
    }
}
